%% Arithmetic

\begin{frame}
  \frametitle{}
  \begin{itemize}
   \item  Конструкция {\tt ((...))}
    \begin{block}{Примеры}
     {\tt (( a=10 )); echo \$(( a++ )); echo \$a; } 
    \end{block}
   \item  {\tt let}
    \begin{block}{Примеры}
     {\tt let a=10; echo \$a; let a+=-2; echo \$a; echo \$(( --a)); echo \$a} 
    \end{block}
   \item  {\tt expr } не нужен
    
  \end{itemize}
\end{frame}
